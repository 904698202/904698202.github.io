Vue3.x

基础

安装

方式一：CDN引入

<script src="https://unpkg.com/vue@next"></script>

方式二：下载完js源代码后再通过script标签引入

<script src="./js/vue3.js"></script>

方式三：npm下载

npm install vue@next

初始化

vue3是使用Vue.createApp(obj)来创建Vue实例，并且通过mount()挂载到指定的元素上

<body>
  <div id="app">
    {{num}}
  </div>
  <script src="https://unpkg.com/vue@next"></script>
  <script>
    //Model数据层
    const obj = {
      data() {
        return {
          num:111
        }
      },
    }
    //创建Vue3实例
    const app = Vue.createApp(obj);
    //挂载在元素上
    app.mount('#app')
  </script>
</body>

简写为：

<body>
  <div id="app">
    {{num}}
  </div>
  <script src="./js/vue3.js"></script>
  <script>
    Vue.createApp({
      data() {
        return {
          num: 111
        }
      },
    }).mount('#app')
  </script>
</body>

## 查看@vue/cli版本，确保@vue/cli版本在4.5.0以上

cd vue_test

组合式API

setup()

Vue3中的一个新的配置项，值是一个函数，组件中所用到的：数据、方法等等，均要配置在setup中。

两种返回值：

  • 若返回一个对象，对象中的属性与方法可以直接在模板中使用
<template>
<h1>个人信息</h1>
<h2>name:{{name}}</h2>
<h2>age:{{age}}</h2>
<button @click="sayHello">sayHello</button>
</template>

<script>

export default {
  name: 'App',
  // 仅仅测试setup，不考虑响应式
  setup() {
    // 属性
    let name = '张三'
    let age = 18
    
    // 方法
    function sayHello() {
      alert(`我叫${name},${age}岁`)
    }

    // 返回值
    return {
      name,
      age,
      sayHello
    }
  }
}
</script>
  • 若返回一个渲染函数，则可以自定义渲染内容（原先模板中定义的内容会直接被渲染函数的内容替换掉）
<template>
<h1>个人信息</h1>
</template>

<script>
// 先引入渲染函数
import { h } from "vue";
export default {
  name: "App",
  setup() {
    return () => {
      return h("h1", "wwh");
    };
  },
};
</script>

注意点：

  • 尽量不要与Vue2的配置混用

    • Vue2配置（data、methods、computed）中可以访问到setup中的属性、方法，但是setup访问不到Vue2配置中的属性方法。
    • 如果有重名，setup优先。
    <template>
  <h1>个人信息</h1>
  <h2>name:{{ name }}</h2><!--name:zhangsan-->
  <h2>age:{{ age }}</h2><!--age:18-->
  <!--如果有重名，setup优先-->
  <h2>gender:{{ gender }}</h2><!--gender:男-->
	
  <h2>phone:{{ phone }}</h2><!--phone:119-->
  <button @click="sayHello">sayHello</button>
  <button @click="sayHi">sayHi</button>
</template>

<script>
export default {
  name: "App",
  data() {
    return {
      name:'zhangsan',
      age:18,
      phone:110
    }
  },
  methods: {
    sayHello() {
      console.log(this)//[proxy]
      //1、vue2的options可以读取setup中返回的内容
      console.log(`${this.name},${this.gender},${this.age}岁,${this.phone}`)//zhangsan,男,18岁,119
    }
  },
  setup() {
    let gender = '男'
    let phone = 119
    function sayHi(){
      console.log(this)//[proxy]
      console.log(gender)//男
      //2、setup中无法读取vue2中options的内容
      console.log(this.name,this.age,this.sayHello)//undefined undefined undefined
    }
    return {
      gender,
      phone,
      sayHi
    }
  },
};
</script>

  • setup不能是一个async函数，因为加上async后返回值不再是我们想要return的对象，而是返回一个promise，模板无法看到该promise的内容。

  • setup中定义的数据不是响应式的。

    <template>
  <h1>个人信息</h1>
  <h2>name:{{ name }}</h2>
  <h2>age:{{ age }}</h2>
  <button @click="change">改变</button>
</template>

<script>
export default {
  name: "App",
  setup() {
    //此时的数据还没有响应式
    let name = "zhangsan";
    let age = 18;
	
    //使用普通方法改变数据，数据会变模板内容不变
    function change() {
      name = "lisi";
      age = 48;
      console.log(name, age);//lisi 48
    }

    return {
      name,
      age,
      change,
    };
  },
};
</script>

ref()

  • 可以使setup中的数据变成一个RefImpl的实例对象，从而实现响应式。

注意：

  • 在setup中，想要操作数据，使模板发生响应式的改变需要先从ref加工后的对象内取出这个value，再进行更改。
  • 在模板中无需通过.value去读取值，vue3自动就会去解析。
  • 当ref()内加工的是一个对象（引用数据类型），想要进行响应式更改时，只需要对此对象名进行.value取值，而对象内部的属性不需要。

总结：

  • ref()处理基本数据类型时，使用的是Object.defineProperty()中的getter()、setter()进行数据劫持。
  • ref()处理引用数据类型时，内部求助了下文的reactive函数，使用的是Proxy。
<template>
  <h1>个人信息</h1>
  <h2>name:{{ name }}</h2>
  <h2>age:{{ age }}</h2>
  <h3>job.type:{{job.type}}</h3>
  <h3>job.salary:{{job.salary}}</h3>
  <button @click="change">改变</button>
</template>

<script>
import { ref } from "vue";
export default {
  name: "App",
  setup() {
    let name = ref("zhangsan");
    let age = ref(18);
    let job = ref({
      type:"前端",
      salary:"30K"
    })
    function change() {
      name.value = "lisi";
      age.value = 48;
      job.value.type = '后端'
      job.value.salary = '15K'
      console.log(name, age);
      /**
       * {
          "__v_isShallow": false,
          "dep": {
              "w": 0,
              "n": 0
          },
          "__v_isRef": true,
          "_rawValue": "zhangsan",
          "_value": "zhangsan"
        }
       */
      console.log(job)
    }

    return {
      name,
      age,
      job,
      change,
    };
  },
};
</script>

reactive()

  • 基于proxy，可以将一个引用数据类型加工成响应式对象。
  • 如果要加工基本数据类型只能使用ref()。
  • 使用reactive()加工后的响应式对象，想操作其中数据不用再.value，可以直接如同JavaScript一样对对象或数组进行操作。
<template>
  <h1>个人信息</h1>
  <h3>job.type:{{job.type}}</h3>
  <h3>job.salary:{{job.salary}}</h3>
  <h3>c:{{job.a.b.c}}</h3>
  <h3>hobby:{{hobby}}</h3>
  <button @click="change">改变</button>
</template>

<script>
import { reactive } from "vue";
export default {
  name: "App",
  setup() {
    let job = reactive({
      type:"前端",
      salary:"30K",
      a:{
        b:{
          c:1
        }
      }
    })

    let hobby = reactive(['抽烟','喝酒','烫头'])

    function change() {
      console.log(job);//Proxy {type: '前端', salary: '30K'}
      job.type = '后端'
      job.salary = '20K'
      job.a.b.c = 2
      hobby[1] = '学习'
    }

    return {
      job,
      hobby,
      change,
    };
  },
};
</script>
  • 当setup中有基本数据类型又有引用数据类型时，可以把他们包裹在一个对象中，一同交给reactive()处理。
<template>
  <h1>个人信息</h1>
  <h3>name:{{ person.name }}</h3>
  <h3>age:{{ person.age }}</h3>
  <h3>job.type:{{ person.job.type }}</h3>
  <h3>job.salary:{{ person.job.salary }}</h3>
  <h3>c:{{ person.job.a.b.c }}</h3>
  <h3>hobby:{{ person.hobby }}</h3>
  <button @click="change">改变</button>
</template>

<script>
import { reactive } from "vue";
export default {
  name: "App",
  setup() {
    let person = reactive({
      name: "zhangsan",
      age: 18,
      job: {
        type: "前端",
        salary: "30K",
        a: {
          b: {
            c: 1,
          },
        },
      },
      hobby: ["抽烟", "喝酒", "烫头"],
    });

    function change() {
      person.name = 'lisi'
      person.job.a.b.c = 22
      person.hobby[1] = '学习'
    }

    return {
      person,
      change,
    };
  },
};
</script>

ref()与reactive()相对比

  • 定义数据的角度
    • ref()多用来定义基本数据类型，reactive()多用来定义引用数据类型（对象、数组……）。
    • ref()也可以用来定义引用数据类型，但是其内部会自动通过reactive()将其转化为代理对象。
  • 原理的角度
    • ref()通过Object.defineProperty()的get()与set()来实现响应式。
    • reactive()通过Proxy来实现响应式，并且通过Reflect来对源对象进行操作。
  • 使用的角度
    • ref()定义的数据在操作时需要.value，在模板中不需要.value。
    • reactive()定义的数据在操作或读取时均不需要.value。

深入

响应式原理

Vue2.x

在Vue2.x中，响应式是通过Object.defineProperty实现的。

// 源数据
let person = {
    name: "zhangsan",
    age: 18
}

// 模拟vue2的响应式
const p = {}
// 对源数据的name进行数据劫持
Object.defineProperty(p, 'name', {
    // 允许被操作
    configurable: true,
    // 当p的name被调用执行
    get() {
        console.log('有人读取了name属性');
        return person.name
    },
    // 当p的name被设置
    set(value) {
        console.log('有人修改了name属性');
        person.name = value
    }
})

Object.defineProperty(p, 'age', {
    // 当p的age被调用执行
    get() {
        console.log('有人读取了age属性');
        return person.age
    },
    // 当p的age被设置
    set(value) {
        console.log('有人修改了age属性');
        person.age = value
    }
})

存在问题：

  • 新增属性、删除属性，界面不会更新，需要通过Vue.set()、this.$set()、Vue.delete()、this.$delete()这些API来完成对响应式对象的新增删除属性操作。
  • 直接通过下标修改数组，界面不会自动更新，需要通过一系列数组中的响应式方法来实现。
  • 使用Object.defineProperty()实现响应式时需要将对象里面每一个属性都进行代理，性能较低。

Vue3.x

在Vue3.x中，响应式是通过Proxy（代理对象）配合Reflect（反射对象）实现的。

  • 通过Proxy，拦截代理对象中任意属性的变化。
  • 通过Reflect，对被代理对象进行相应的操作。

我们需要通过实例化Proxy对象来实现对另一个对象的代理，该对象有两个参数。

参数 作用
target 被代理的任意类型的对象，包括原生数组、对象，甚至另一个
handler 通常为一个对象，对象中有许多方法用于操作被代理的目标对象

Reflect并非一个构造函数，不能用new进行实例化，因此其内部方法都是静态方法，方法与Object上面的方法有所类似，可以看作是Object上静态方法的移植。

使用Reflect进行对代理对象的操作时，会有一个boolean型返回值，可以通过这个返回值判断该方法是否定义成功，而不是像Object一样直接报错，需要使用许多try...catch...去捕获错误，才能保证代码的继续运行。

// 源数据
let person = {
    name: "zhangsan",
    age: 18
}

// 模拟vue3的响应式
const p = new Proxy(person,{
    // 常用两个参数1、target被代理的对象 2、propName代理对象被操作的属性名
    get(target,propName) {
        console.log(`有人读取了p身上的${propName}属性`,target,propName);
        return Reflect.get(target,propName)
    },
    // 比起get()多一个对象为操作的属性值
    set(target,propName,value) {
        console.log(`有人设置了p身上的${propName}属性`,target,propName,value)
        Reflect.set(target,propName,value)
    },
    // 用于进行对被代理对象的删除
    deleteProperty(target,propName) {
        console.log(`有删除了p身上的${propName}属性`,target,propName);
        return Reflect.deleteProperty(target,propName)
    }
})
