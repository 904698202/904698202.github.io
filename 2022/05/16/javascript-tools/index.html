自定义工具库

函数相关

call()

作用：改变函数的this指向。

参数：

  1. 待执行的函数
  2. this的目标指向对象
  3. 传入多个参数给待执行函数进行调用

返回值：改变this指向后函数执行的值

实现思路：将要改变的方法设置为this目标对象的一个临时方法，通过执行这个对象上的临时方法来改变this指向。

代码：

function call(Fn, obj, ...args) {
  // 判断当obj为空时指向全局对象
  if (obj === undefined || obj === null) {
    obj = globalThis
  }
  // 1、为传入的对象添加临时方法
  obj.temp = Fn
  // 2、传入参数，执行此对象上的方法
  let result = obj.temp(...args)
  // 3、删除此对象上的方法
  delete obj.temp
  // 4、将方法执行的结果返回
  return result
}

调用方法：

function add(a, b) {
  console.log(this);
  return a + b + this.c
}
// 为对象obj添加属性
let obj = {
  c: 521
}
// 为全局对象添加属性
window.c = 1314

console.log(call(add, obj, 10, 20));//Object 551
console.log(call(add, null, 10, 20));//Window 1344

apply()

作用：改变函数的this指向。

参数：

  1. 待执行的函数
  2. this的目标指向对象
  3. 传入一个数组给待执行函数进行调用

返回值：改变this指向后函数执行的值

实现思路：与call()的实现思路一致，只不过给待调用函数传入的参数是一个数组。

代码：

function apply(Fn, obj ,args) {
  // 判断对象是否指定
  if (obj === undefined || obj === null) {
    obj = globalThis
  }
  // 1、为对象添加临时方法
  obj.temp = Fn
  // 2、传入参数，通过此对象执行方法
  let result = obj.temp(...args)
  // 3、删除对象中的临时方法
  delete obj.temp
  // 4、返回结果
  return result
}

调用方法：

function add(a,b) {
  console.log(this);
  return a +b +this.c
}
// 为对象obj添加属性
let obj = {
  c:521
}
// 为全局对象添加属性
window.c = 1314

console.log(apply(add,obj,[10,20]));//Object 551
console.log(apply(add,null,[10,20]));//Window 1344

bind()

作用：改变函数执行的this指向，将改变完对象的函数作为返回值保存，以供多次调用。

参数：

  1. 待执行函数
  2. this的目标指向对象
  3. 传入一个或多个值给待执行函数进行调用

返回值：返回改变this指向后的函数

实现思路：返回一个function，这个function也可以接收一系列参数，并且返回上方将这个参数与原来bind()接收的参数一起传给call()去改变this指向并且返回call()执行后的结果。

代码：

function bind(Fn,obj,...args){
  // 1、返回一个函数
  // 2、改变指向时没有传入参数，可以在调用改变了指向后的方法再传入参数
  return function(...args2) {
    return call(Fn,obj,...args,...args2)
  }
}

调用方法：

function add(a,b) {
  console.log(this);
  console.log(arguments);
  return a +b +this.c
}
// 为对象obj添加属性
let obj = {
  c:521
}
// 为全局对象添加属性
window.c = 1314

let fn = bind(add,obj,10,20)
console.log(fn());
//{c: 521, temp: ƒ}
//Arguments(2) [10, 20...]
//551

let fn2 = bind(add,obj)
console.log(fn2);
/**
ƒ (...args2) {
    return call(Fn,obj,...args,...args2)
  }
*/
console.log(fn2(20,20));
//{c: 521, temp: ƒ}
//Arguments(2) [20, 20...]
//561

throttle()

作用：节流函数——使一个回调函数在一段时间内最多执行一次。

适用场景：浏览器下拉事件触发的回调…

参数：

  1. 要节流的函数
  2. 每次执行的间隔

返回值：一个用于执行的函数。

实现思路：在节流函数内定义一个开始时间，在返回的函数内获取当前时间，当当前时间与事件触发的时间差大于等待时间则用call()去执行要节流的函数，并且更新开始时间。

代码：

function throttle(callback, wait) {
  // 定义开始时间
  let start = 0
  // e为事件对象
  return function(e){
    // 获取当前时间戳
    let now = Date.now()
    // 判断
    if (now - start >= wait) {
      // 满足条件则执行回调
      // 函数返回出去一定会放在外面的回调函数的位置
      // 因此只需要用原本的this去指向他
      callback.call(this,e)
      start = now
    }
  }
}

调用方法：

//绑定一个屏幕滚动的响应函数，用节流函数将这个回调包裹起来
window.addEventListener('scroll',throttle(function(e){
  console.log(Date.now(),e);
},500))

debounce()

作用：防抖函数——当上一个事件触发，不立即执行事件绑定的回调函数，而是等待一段时间。如果在这等待的时间内此事件再次触发，则停止上一个事件的回调，执行新事件的回调。

适用场景：输入框校验、搜索…

参数：

  1. 需要防抖的回调函数
  2. 等待的时间

返回值：一个用于执行的函数

实现思路：利用定时器，先在防抖函数内定义一个定时器遍历；然后在返回的函数内判断这个定时器变量是否为null，如果不为null，则清空上一个定时器，接着开启新的一个定时器；当定时器被延时触发，则使用call()调用传来的回调，并且清空定时器变量表示这一个事件的回调已经执行完毕。

代码：

function debounce(callback,time) {
  // 定时器变量
  let timer = null
  // 返回一个函数，需要接收事件源作为回调函数执行的参数
  return function(e) {
    // 当执行新的函数时，若发现有正在等待的回调定时器，则停止该定时器
    if (timer !== null) {
      clearTimeout(timer)
    }
    // 延迟触发事件回调
    timer = setTimeout(() => {
      callback.call(this,e)
      // 每次执行完都需要清空定时器变量
      // 否则变量一直保存的话会一直触发清空定时器执行无用代码
      timer = null
    }, time);
  }
}

调用方法：

<input type="text">
<script>
  const input = document.querySelector('input')
  //使用防抖函数将事件的回调包裹起来
  input.onkeydown = debounce(function(e) {
    console.log(e.keyCode);
  },1000)
</script>

数组相关

map()

作用：传入一个数组，将数组中每一个元素在回调函数中处理后返回一个新的数组。

参数：

  1. 待处理数组
  2. 处理数组的回调函数（此回调接收原数组的值及其下标值）

返回值：处理后的数组

实现思路：定义一个存放结果的数组，遍历待处理数组，将每个数组元素传入回调函数处理后，将回调函数的返回值放入结果数组中并且返回结果数组。

代码：

function map(arr,callback) {
  // 用于存放结果数组
  let result = []
  // 循环遍历原数组中的每一个元素，将其传入回调函数中执行
  // 将回调函数的结果保存在result中并返回
  for(let i=0;i<arr.length;i++){
    result.push(callback(arr[i],i))
  }
  return result
}

调用方法：

const arr = [1,2,3.4,5]
const result = map(arr,(item,index) => {
  return item*10
})

console.log(result);//[10, 20, 34, 50]

reduce()

作用：传入一个待执行数组，将此数组每一个元素放入回调函数中执行，返回累加的执行结果以及初始值的和。

参数：

  1. 待执行数组
  2. 处理数组的回调函数（此回调接收自定义的初始值或者上一次回调的结果值和一个新的值）
  3. reduce()统计的初始值

返回值：数值经过回调函数处理后的结果之和

实现思路：定义一个结果变量，此变量初始值为reduce()的初始值，遍历待执行数组，将结果变量以及每个数组元素传入回调，此回调的第一个参数为初始值或者上一个函数的返回值，第二个参数为当前要处理的数组元素 ，循环完成后返回结果变量。

代码：

function reduce(arr,callback,initVal=0) {
  // 定义变量存放结果，初始值为函数外指定的值
  let result = initVal
  // 遍历数组元素进行操作
  // 调用回掉后应该改变结果里面的数
  for(let i=0; i<arr.length ; i++) {
    result = callback(result,arr[i])
  }
  // 返回结果
  return result
}

调用方法：

const arr = [1,2,3,4,5]
let result = reduce(arr,function(preValue,value){
	return preValue + value
})
console.log(result);//15

filter()

作用：传入一个待处理数组，当数组中的元素传入回调函数中得到的结果是真时，则将这个元素放入结果数组中返回。

参数：

  1. 待执行数组
  2. 处理数组元素的回调

返回值：满足回调条件的新数组

实现思路：定义一个结果数组，将数组每一个元素放入回调中执行，若结果为真则将此元素放入结果数组中返回。

代码：

function filter(arr,callback) {
  // 用于存放结果的数组
  const result = []
  // 循环数组中每个元素进行回调的操作
  for(let i=0;i<arr.length;i++){
    // 若回调结果为真压入filter操作结果的数组
    if (callback(arr[i])) {
      result.push(arr[i])
    }
  }
  // 返回结果数组
  return result
}

调用方法：

const arr = [1,2,3,4,5]
const result = filter(arr,(item,index) => item % 2 !=0)
console.log(result)//[1, 3, 5]

find()

作用：在数组中返回满足回调的第一个数组元素，若没找到则返回undefined。

参数：

  1. 寻找元素的目标数组
  2. 处理数组元素的回调

返回值：

  • 满足回调的第一个数组元素下标
  • undefined

实现思路：遍历传入的目标数组，将数组元素及其下标放入回调中执行。若回调执行结果为true则返回该数组元素；若没有满足回调的元素存在则返回undefined。

代码：

function find(arr,callback) {
  // 遍历数组
  for (let i = 0; i < arr.length; i++) {
    // 在数组中返回满足回调的第一个数组元素
    if(callback(arr[i],i)) return arr[i]
  }
  // 若没有找到返回undefined
  return undefined
}

调用方法：

const arr = [1,2,3,4,5,6]
const result = find(arr,(item,index) =>{
    return item>4
})

console.log(result);//5

findIndex()

作用：在数组中返回满足回调的第一个数组元素的下标，若没有找到返回-1。

参数：

  1. 寻找元素的目标数组
  2. 处理数组元素的回调

返回值：

  • 满足回调的第一个数组元素的下标
  • -1

实现思路：遍历目标数组，将目标数组的每一个元素及其下标放入回调中执行。若回调结果为true，返回该数组元素下标；若回调结果为false，返回-1。

代码：

function findIndex(arr,callback) {
  // 遍历数组
  for (let i = 0; i < arr.length; i++) {
    // 在数组中返回满足回调的第一个数组元素下标
    if(callback(arr[i],i)) return i
  }
  // 若没有找到返回-1
  return -1
}

调用方法：

const arr = [1,2,3,4,5,6]
const result = findIndex(arr,(item,index) =>{
    return item > 4
})

console.log(result);//4

every()

作用：检测数组中元素是否全部满足回调条件。

参数：

  1. 待检查数组
  2. 处理数组元素的回调

返回值：Boolean

实现思路：遍历待检查数组，若有一个数组元素不满足回调条件则返回false，否则返回true。

代码：

function every(arr,callback) {
  for (let i = 0; i < arr.length; i++) {
    if (!callback(arr[i],i)) return false
  }
  return true
}

调用方法：

const arr = [1,2,3,4,5,1]
const result = every(arr,(item,index) => {
  return item>0
})
console.log(result);//true

some()

作用：检测数组元素中是否有元素满足回调条件。

参数：

  1. 待检测数组
  2. 处理数组条件的回调

返回值：Boolean

实现思路：遍历待检查数组，若有一个数组元素满足回调条件则返回true，否则返回false。

代码：

function some(arr,callback) {
  for (let i = 0; i < arr.length; i++) {
    if (callback(arr[i],i)) return true
  }
  return false
}

调用方法：

const arr = [1,2,3,4,5,1]
const result = some(arr,(item,index) => {
  return item>4
})
console.log(result);//true

unique()

作用：数组去重。

参数：待处理数组

返回值：去重后的新数组。

实现思路：

  1. forEach + indexOf 双重遍历——定义一个新的结果数组，遍历带处理数组中每一个元素，如果在结果数组中没有找到正在遍历的元素值则将这个元素放入结果数值中返回。
  2. forEach + 对象 一次遍历——定义一个空的结果数组以及一个空的结果对象，遍历待处理数组，以数组值为结果对象的key，以Boolean值为结果对象的value作为标记。若没有在此对象中找到该元素的标记则在对象中以该元素为key，以true为value进行标记，并且放入结果数组中返回。
  3. spread拓展运算符+Set()数据结构——以Set()数据结构中元素值不重复的特性，将数组放入其中去重，并且用spread拓展运算符展开这个Set()然后以数组的方式返回结果。

代码：

//forEach + indexOf 双重遍历
function unique1(arr) {
  // 定义一个结果数组
  const result = []
  // 遍历传入数组的每个元素
  arr.forEach(item => {
    // 若结果数组中没有该元素则把他放入结果数组
    if(result.indexOf(item) === -1) result.push(item)
  });
  return result
}

//forEach + 对象 一次遍历
function unique2(arr) {
  // 结果数组
  const result = []
  // 空结果对象
  const obj = {}
  // 遍历数组
  arr.forEach(item => {
    // 当对象中key为此元素的值不存在时，添加他的值为true，然后放入结果数组
    // 在者之后就能通过判断该对象的值是不是为true来决定要不要把此元素放入结果数组
    if(obj[item] === undefined) {
      obj[item] = true
      result.push(item)
    }
  })
  return result
}

//ES6语法：... + Set
function unique3(arr) {
  return [...new Set(arr)]
}

调用方法：

const arr = [1,1,2,4,5,1,3,4]
console.log(unique1(arr));//[1, 2, 4, 5, 3]
console.log(unique2(arr));//[1, 2, 4, 5, 3]
console.log(unique3(arr));//[1, 2, 4, 5, 3]

concat()

作用：合并数组。

参数：

  1. 基础数组
  2. 要合并的一个数组或一个数（用rest参数接收）

返回值：合并后的新数组

实现思路：将基础数组使用spread展开后作为结果数组的基础元素，遍历rest接收的参数。判断这个参数是不是一个数组，若是数组，则用spread展开后再调用push()放入结果数组；若只是一个值，则直接push()放入结果数组。

代码：

function concat(arr,...args) {
  const result = [...arr]
  args.forEach(item => {
    Array.isArray(item)?result.push(...item):result.push(item)
  });
  return result
}

调用方法：

const arr = [1,2,4]
console.log(concat(arr,1,[2,3],4))//[1, 2, 4, 1, 2, 3, 4]

slice()

作用：数组截取。

参数：

  1. 要截取的目标数组
  2. 要截取的数组的起始下标
  3. 要截取的数组的结束下标（该下标所在值不会被截取）

返回值：返回截取出来的结果数组

实现思路：遍历要截取的目标数组，当其下标在起始下标和结束下标的这个范围内，将此元素放入结果数组中返回。需要通过判断对传入的值进行特殊处理。

代码：

function slice(arr,start,end) {
  // 定义结果数组
  const result = []

  // 当没有传入数组时直接返回空数组
  if(arr.length === 0) return []

  // 当没有传入start时默认从0开始截取
  // start = start || 0
  start ||= 0
  // 当截取的开头大于等于数组的长度，返回空数组
  if(start >= arr.length) return []

  // 当没有传入end时，默认截取开始之后的所有值
  // end = end || arr.length
  end ||= arr.length
  // 当传入的结束值小于开始值，结束值默认为数组长度
  if(end < start) {
    end = arr.length
  }

  // 循环数组
  for (let i = 0; i < arr.length; i++) {
    // 当数组下标在截取范围内则放入结果数组
    if (i >= start && i < end) {
      result.push(arr[i])
    }
  }

  // 返回结果数组
  return result
}

调用方法：

const arr = [1,2,6,3,4,5,6,7,8]
const result = slice(arr,1)
console.log(result);//[2, 6, 3, 4, 5, 6, 7, 8]

flat()

作用：数组扁平化。

参数：待处理数组

返回值：返回一个新的结果数组

实现思路：

  1. 递归 + concat()——定义一个新的结果数组，遍历待处理数组。若数组元素是一个数组，则进行递归调用；若不是，则直接将该元素合并入数组。
  2. some() + concat() + 拓展运算符——将数组复制一份作为基础结果，用while循环判断结果数组中是否存在数组，若存在，则将这个元素展开后合并入结果数组。

代码：

//递归 + concat()
export function flat1(arr) {
  const result = []
  // 遍历数组
  arr.forEach(item => {
    // 当数组中元素是一个数组时，递归调用此方法展开数组
    // 当数组中元素不是一个数组时，将该元素合并到结果数组中
    result = Array.isArray(item)?result.concat(flat1(item)):result.concat(item)
  });
  return result
}

//some() + concat() + 扩展运算符
export function flat2(arr) {
  // 把数组复制一份新的用于操作
  const result = [...arr]

  //循环当结果的数组中存在一个元素是数组的话，用拓展运算符展开后合并不是数组的那一部分元素
  while (result.some(item => Array.isArray(item))) {
    result = [].concat(...result)
  }
  return result
}

调用方法：

const arr = [1,2,[3,4,[5,6],7],8]
console.log(flat1(arr));//1, 2, 3, 4, 5, 6, 7, 8
console.log(flat2(arr));//1, 2, 3, 4, 5, 6, 7, 8

chunk()

作用：数组分块—按照第二个参数，将数组分成若干个数组组成的二维数组。

参数：

  1. 待分块数组
  2. 每一块的大小

返回值：分块后的二维数组

实现思路：定义一个结果数组与一个临时的块数组，遍历待分块数组，先把空的临时数组块放入结果数组中，然后不断往这个数组块中放入元素，如果数组块的大小到达了我们设定的大小则开启一个新的数组块再放入结果中进行循环。

代码：

function chunk(arr,size=1) {
  // 用于存放结果
  const result = []
  // 用于存放结果中的其中一个数组
  const temp = []
  // 遍历数组
  arr.forEach(item => {
    // 先给结果数组中放入一个空的数组
    if (temp.length === 0) {
      result.push(temp)
    }
    // 不断地往结果中的这个数组中压入元素
    temp.push(item)
    // 压入过后判断此数组是否饱和
    // 饱和了则开启一个新的数组块
    if (temp.length === size) {
      temp = []
    }
  });

  return result
}

调用方法：

const arr = [1,2,3,4,5,6,7]
console.log(chunk(arr,3));//[[1,2,3],[4,5,6],[7]]

difference()

作用：数组差集——判断第一个数组内的元素是否存在于第二个数组；若存在，剔除该元素，返回一个新的数组。

参数：

  1. 基础数组
  2. 排除名单数组

返回值：求差集后的新数组

实现思路：使用filter()过滤数组，使用includes()方法0判断基础数组中的元素是否存在于排除名单数组中。

代码：

function difference(arr1,arr2) {
  // 数组一为空，返回空数组
  if (arr1.length === 0) {
    return []
  }

  // 数组二为空，通过slice返回一个新的arr1数组
  if (arr2.length === 0) {
    return arr1.slice()
  }

  // 数组一的元素若不存在于数组二则放入结果数组返回
  return arr1.filter(item => !arr2.includes(item))
}

调用方法：

const arr1 = [1,2,3,5,7]
const arr2 = [5,8]
console.log(difference(arr1,arr2));//[1,2,3,7]

pull()

作用：删除数组元素。

参数：

  1. 待删除的数组
  2. 要删除的值

返回值：被删除的元素的数组

实现思路：创建一个结果数组用于存放被删除的数，对待删除的数组进行遍历，当遍历的元素存在于rest参数形成的数组中时，将这个元素放入被删除的数组中，并且在原数组中删除这个数，注意被删除了一位后下标需要自减一位，以免跳过了被删除元素的下一位的判断。

代码：

function pull(arr,...args) {
  // 存放被删除的数
  const result = []
  // 遍历数组元素
  for (let i = 0; i < arr.length; i++) {
    // 此元素书否存在于传入的参数中
    if (args.includes(arr[i])) {
      // 若存在先保存该数组
      result.push(arr[i]);
      // 调用splice()从原数组中删除这个数
      arr.splice(i,1);
      // 由于删除了一个数，所以下次遍历时下标要后移动一位
      // 否则会跳过一个数的判断
      i--;
    }
  }

  // 返回被删除的元素数组
  return result
}

调用方法：

const arr = [1,2,3,3,5,7]
console.log(pull(arr,3,7));//[3, 3, 7]
console.log(arr);//[1, 2, 5]

newInstance()

作用：创建新的对象。

参数：

  2. 创建对象时要传入的参数

返回值：

  • 新的实例对象

function newInstance(Fn,...args) {

  // 如果不是一个对象，就返回构建的实例对象
  return result instanceof Object ? result : obj
}

instanceOf()

作用：比对对象类型——通过判断对象的隐式原型是否存在于构造函数的原型链上。

参数：

  1. 要判断的实例对象
  2. 被判断的构造函数

返回值：Boolean

实现思路：获取实例对象的隐式原型与构造函数的显式原型，通过遍历原型链的方式来比对对象类型。

代码：

function instanceOf(obj , Fn) {
  // 获取实例对象的隐式原型
  let proto = obj.__proto__
  // 获取构造函数的显式原型
  let prototype = Fn.prototype
  // 遍历原型链
  while (proto) {
    // 当原型对象相等返回true
    if (prototype === proto) {
      return true
    }
    // 沿着隐式原型向上
    proto = proto.__proto__
  }
  // 原型对象匹配不成功返回false
  return false
}

调用方法：

let p = new P
console.log(instanceOf(p,P));

mergeObject()

作用：传入多个对象，将它们合成一个对象并返回

参数：使用rest参数接收多个对象

返回值：合并后的新对象

实现思路：定义一个新对象用于存放结果，遍历传入的对象当结果对象中出现与正在遍历的对象有重名的key则用一个数组去合并这两个值；如果没有重名则直接将对象的值赋予结果对象。

代码：

function mergeObject(...args) {
  // 用于存放新的对象
  let result = {}
  // 遍历传入参数，获取所有对象
  args.forEach(obj => {
    // 取到一个对象中所有的key值，并遍历这些key
    Object.keys(obj).forEach(key => {
      // 当结果对象中出现重名的key，用一个新的数组去保存他们合并的结果
      if (result.hasOwnProperty(key)) {
        result = [].concat(result[key],obj[key])
      } else {
        // 如果在结果对象中没有出现重名的key值，则直接将此对象的值放入结果数组
        result[key] = obj[key]
      }
    })
  });
  // 返回结果对象
  return result
}

调用方法：

  a:[
    { x:2 },
    { y:4 }
  ],
  b:1
}

const other = {
  a:{ z:3 },
  b:[2,3],
  c:'foo'
}

console.log(mergeObject(object,other));

clone()

作用：浅拷贝（引用地址拷贝）——拷贝一个新的结果（修改了拷贝出来的对象的值原对象的值也会跟着改变）

参数：被克隆的目标

返回值：克隆后的对象

实现思路：

  • 使用ES6拓展运算符实现——判断当前是否为引用数据类型，若不是则直接返回该结果；若是则再次判断其是否是一个数组后分别进行复制。
  • 使用ES5语法实现——判断当前诗句是否为引用数据类型，若不是则直接返回该结果；若是则根据被克隆的目标的类型来创建一个空的数组或对象。遍历目标上的键，将目标当前身上的属性的键值赋值一份给结果变量。

代码：

// 使用ES6拓展运算符实现
function clone1(target) {
  // 判断当前目标是不是引用数据类型（由于typeof的特点需要排除null）
  if (typeof target === 'object' && target !== null) {
    // 数组也是引用数据类型的一种，要拿出来单独处理
    if (Array.isArray(target)) {
      return [...target]
    } else {
      return {...target}
    }
  } else {
    // 如果是基本数据类型则直接返回
    return target
  }
}

// ES5语法实现
function clone2(target) {
  // 判断数据类型
  if (typeof target === 'object' && target !== null) {
    // 根据target的数据类型来创建初始值
    const result = Array.isArray(target) ? [] : {}
    // 遍历target的key
    for (let key in target) {
      // 由于for...in..也能遍历对象原型上的数据
      // 因此需要判断对象当前身上是否包含该属性
      if (target.hasOwnProperty(key)) {
        // 将属性设置到result中
        result[key] = target[key]
      }
    }
    return result
  } else {
    return target
  }
}

    m:1
  }
}

deepClone()

作用：深拷贝（值拷贝）——对拷贝后的对象进行修改不会影响原先的数据。

参数：被克隆的目标

返回值：克隆后的对象

实现思路：

  1. 乞丐版：使用JSON字符串转换对象返回新的拷贝对象，缺点是不能拷贝对象中的方法，也不能处理循环引用的问题。
  2. 普通版：在ES5浅拷贝的基础上，使用递归的方式进行深拷贝，缺点是如果遇到了循环引用会发生死循环导致栈内存的溢出。
  3. 完整版：在普通版的基础上，通过设置map()的方式来解决循环引用的问题。
  4. 性能优化版本：在完整版的基础上，使用forEach代替了for…in…的循环——for…in…会遍历到对象的原型上的内容，还需要多加一次判断，因此性能较差。

代码：

//乞丐版
function deepClone1(target) {
  let str = JSON.stringify(target)
  return JSON.parse(str)
}

//普通版
function deepClone2(target) {
  // 判断传入的值是不是对象
  if (typeof target === 'object' && target !== null) {
    // 根据传入的对象的类型创建一个存放结果的容器
    const result = Array.isArray(target) ? [] : {}
    // 通过key来循环传入的对象中的key值
    for(let key in target) {
      // 如果该对象的key值是该对象本身的属性（不是从原型链上获取到的）
      if (target.hasOwnProperty(key)) {
        // 将对象的key对应的value放到结果中对应的位置
        // 由于传入的对象的value可能是一个数组或对象，因此要进行递归调用
        //如b:['e','f','g']
        result[key] = deepClone2(target[key])
      }
    }
    // 将结果返回
    return result
  } else {
    // 当传入的值是一个常量，直接返回：如：a:1
    return target
  }
}

//完整版
function deepClone3(target,map = new Map()) {
  if (typeof target === 'object' && target !== null) {
    // 在map中获取此target对应的值作为缓存
    let cache = map.get(target)
    // 当缓存存在就直接把这个缓存返回，不让函数进入循环
    if (cache) {
      return cache
    }
    const result = Array.isArray(target) ? [] : {}
    // 将这个target作为key，result作为value放入map中保存
    // 这个result先是空的，若后面的代码修改了这个result，map中的也会跟着修改
    map.set(target,result)
    for(let key in target) {
      if (target.hasOwnProperty(key)) {
        // 将这个map也作为一个值传入递归调用
        result[key] = deepClone3(target[key],map)
      }
    }
    return result
  } else {
    return target
  }
}

//性能优化
function deepClone4(target,map = new Map()) {
  if (typeof target === 'object' && target !== null) {
    let cache = map.get(target)
    if (cache) {
      return cache
    }
    const isArray = Array.isArray(target)
    const result = isArray ? [] : {}
    map.set(target,result)
    // 判断这个传入的值是数组还是对象
    if (isArray) {
      // 用数组下标去保存结果
      target.forEach((item,index) => {
        // item === target[index]
        result[index] = deepClone4(item,map)
      });
    } else {
      // 先获取所传对象的所有key值，再进行递归赋值
      Object.keys(target).forEach(key => {
        result[key] = deepClone4(target[key],map)
      });
    }
    return result
  } else {
    return target
  }
}

调用方法：

  a:1,
  b:['e','f','g'],
  c:{ h: 20 },
  d:function(){}
}
// 循环引用
obj.b.push(obj.c)
obj.c.j = obj.b

// const result = deepClone1(obj)
// const result = deepClone2(obj)
// const result = deepClone3(obj)

console.log(obj);

事件总线

我们通过定义一个对象作为事件总线，并且在其中有一个callbacks对象属性，该对象属性的key值为事件的类别，value值为这一事件类别的回调函数所形成的数组。

let eventBus = {
  callbacks:{
    // [type]:[fn1,fn2]
  }
}

并且该对象会对外开放三个API：

  1. on：绑定事件
  2. emit：触发事件
  3. off：解绑事件

on

作用：绑定事件——可以为一个类型的事件绑定多个回调函数。

参数：

  1. 事件种类
  2. 事件回调

返回值：无

实现思路：判断传入的事件类型在对象的回调属性内是否已经保存了同样类型的回调，如果存在则将回调函数放入事件总线同一类型的回调形成的数组中；若不存在则在事件总线的回调属性内创建类型并且把回调函数作为这一回调类型数组的初始值。

代码：

eventBus.on = function(type,callback) {
  if (this.callbacks[type]) {
    // 当此类型事件存在，直接把回调函数压入保存
    this.callbacks[type].push(callback)
  } else {
    // 当此类型事件不存在，创建类型，并把回调函数作为初始值
    this.callbacks[type] = [callback]
  }
}

调用方法：

eventBus.on('login',data => {
    console.log(data + '用户已经登录');
})

eventBus.on('login',data => {
    console.log(data + '登录数据已经写入');
})

emit

作用：触发事件——可以触发同一类型事件的所有回调函数

参数：

  1. 事件种类
  2. 事件回调要用到的数据

返回值：无

实现思路：判断传入的事件类型是否存在于事件总线的回调属性中，并且该事件类型是否存在响应的响应回调，若存在则将这个事件对应的回调数组进行遍历执行。

代码：

eventBus.emit = function(type,data){
  // 判断此类型的事件是否存在回调函数
  if (this.callbacks[type] && this.callbacks[type].length > 0) {
    // 若存在则遍历执行
    this.callbacks[type].forEach(callback => {
      callback(data)
    })
  }
}

调用方法：

eventBus.emit('login','张三')

off

作用：解绑事件——将事件类型及其回调函数从对象中删去。

参数：事件类型名称

返回值：无

实现思路：判断是否有传入参数，如果有传入参数则删除指定事件类型的回调，如果没有传入参数则清空所有事件的回调。

代码：

eventBus.off = function(eventName) {
  if (eventName) {
    // 当传入事件名，删除对应事件回调
    delete this.callbacks[eventName]
  } else {
    // 没有传入事件名则清空所有事件回调
    delete this.callbacks
  }
}

调用方法：

eventBus.off('login')

发布订阅模式

发布订阅模式与上述事件总线差不多。也是通过一个对象来保存事件的种类与其对应的回调函数。

const PubSub = {
  // 设置一个唯一的值作为频道标识符
  id:1,
  // 频道与回调保存容器
  callbacks:{
    // channel:{
    //   token_1:callback1,
    //   token_2:callback2
    // }
  }
}

并且这个对象会对外开放三个API：

  1. subscribe：订阅频道
  2. publish：发布消息
  3. unsubscribe：取消订阅

subscribe

作用：订阅频道——可以为一个频道用对象键值对的形式保存一到多个回调。

参数：

  1. 频道名
  2. 回调函数

返回值：无

实现思路：借由发布订阅模式对象内的id生成唯一的频道标识符，并且判断传入的频道是否存在。若存在，则向这个频道中加入key做为唯一标识符的回调；若不存在，则先创建频道，再将唯一标识符与对应回调作为键值对一起保存到此频道对象中。

代码：

PubSub.subscribe = function (channel, callback) {
  // 自动生成频道标识符
  let token = 'token_' + this.id++
  // 当此频道存在，向频道中加入回调函数
  if (this.callbacks[channel]) {
    this.callbacks[channel][token] = callback
  } else {
    // 当频道不存在，先创建频道，再保存回调
    this.callbacks[channel] = {
      [token] : callback
    }
  }
}

调用方法：

    console.log('商家接到订单开始制作',data);
})

    console.log('骑手接到订单准备开始取餐',data);
})

publish

作用：发布消息——给一个频道内所有的回调函数发布消息（传递数据），并且执行频道内所有的回调。

参数：

  1. 频道名
  2. 消息数据

返回值：无

实现思路：判断传入的频道名是否存在，如果存在则先用Object.values取得此频道中所有的回调函数然后依次传入消息数据并执行。

代码：

PubSub.publish = function (channel, data) {
  // 当此频道存在，将信息发给频道中所有回调去执行
  if (this.callbacks[channel]) {
    Object.values(this.callbacks[channel]).forEach(callback => {
     callback(data)
    })
  }
}

调用方法：

PubSub.publish('pay',{
    title:'鱼香肉丝盖饭',
    price:20,
    pos:'三年二班'
})
